详细用法和潜能计算器一样
可用选项 动作栏  进入战斗后手动点击开启
blockedSkills  = '隐杀|定影' 是屏蔽技能 自动出招的时候 不使用
prioritySkills 是优先技能

<red>自动出招-关</red>  <script> (function() {     let blockedSkills = '隐杀|定影';     let prioritySkills = '重击|斩杀';      function init() {         try {             let autoAttackInterval = null;             const combatPanel = document.querySelector('.combat-panel');             if (!combatPanel) return;             const skillContainer = combatPanel.querySelector('.combat-commands');             if (!skillContainer) return;              function getAllSkills() {                 return Array.from(skillContainer.querySelectorAll('.pfm-item'));             }              function getAvailableSkills() {                 const allSkills = getAllSkills();                 const available = allSkills.filter(skill => {                     const style = skill.getAttribute('style') || '';                     return style === '' || style.includes('background-size: 0% 100%');                 });                  const blockedList = blockedSkills.split('|').map(s => s.trim()).filter(Boolean);                 const priorityList = prioritySkills.split('|').map(s => s.trim()).filter(Boolean);                                  const filtered = available.filter(skill => {                     const text = skill.textContent.trim();                     return !blockedList.includes(text);                 });                  if (priorityList.length === 0) return filtered;                  const prioritySkillsAvailable = filtered.filter(skill =>                      priorityList.includes(skill.textContent.trim())                 );                 const normalSkills = filtered.filter(skill =>                      !priorityList.includes(skill.textContent.trim())                 );                  return [...prioritySkillsAvailable, ...normalSkills];             }              const allElements = document.getElementsByTagName('*');             let targetEl = null;             for (let el of allElements) {                 const text = el.textContent ? el.textContent.trim() : '';                 if (text === '自动出招-关') {                     targetEl = el;                     break;                 }             }             if (!targetEl) return;              function startAutoAttack() {                 let currentIndex = 0;                 autoAttackInterval = setInterval(() => {                     const availableSkills = getAvailableSkills();                     if (availableSkills.length === 0) return;                      if (currentIndex >= availableSkills.length) currentIndex = 0;                     const currentSkill = availableSkills[currentIndex];                                          setTimeout(() => {                         currentSkill.click();                     }, 50);                      currentIndex++;                 }, 200);             }              function stopAutoAttack() {                 if (autoAttackInterval) {                     clearInterval(autoAttackInterval);                     autoAttackInterval = null;                 }             }              function toggle(element) {                 try {                     const text = element.textContent.trim();                     let newTag, newText;                     if (text === '自动出招-关') {                         newTag = 'hig';                         newText = '自动出招-开';                         startAutoAttack();                     } else if (text === '自动出招-开') {                         newTag = 'red';                         newText = '自动出招-关';                         stopAutoAttack();                     } else return;                      const newEl = document.createElement(newTag);                     newEl.textContent = newText;                     newEl.style.cursor = 'pointer';                     newEl.addEventListener('click', () => toggle(newEl));                      if (element.parentNode) {                         element.parentNode.replaceChild(newEl, element);                     }                 } catch (err) {}             }              targetEl.style.cursor = 'pointer';             targetEl.addEventListener('click', function() {                 try {                     toggle(this);                 } catch (err) {}             });         } catch (err) {}     }      if (document.readyState === 'loading') {         document.addEventListener('DOMContentLoaded', init);     } else {         init();     } })(); </script>
