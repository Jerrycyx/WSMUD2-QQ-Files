详细用法和潜能计算器一样
可用选项 选择地图  参数写 name(扬州) 

实际功能为 点击进入副本 增加一个一键扫荡你现在拥有的精力的上限次数

<script> try {     const containerSelector = '.item-commands';     const jingliPattern = /当前精力：(\d+)/;      function getSweepCount() {         const allText = document.body.innerText;         const match = allText.match(jingliPattern);         if (match) {             const jingli = parseInt(match[1], 10);             return Math.max(0, Math.floor(jingli / 10));         }         return 0;     }      function updateContainer(container) {         const count = getSweepCount();         if (count <= 0) return;          const sweepOnce = container.querySelector('span[cmd]');         if (!sweepOnce) return;          const originalCmd = sweepOnce.getAttribute('cmd');         const cmdParts = originalCmd.split(' ');         let targetCmd;          if (cmdParts.length === 2) {             targetCmd = `${originalCmd} 0 ${count}`;         } else if (cmdParts.length >= 3) {             cmdParts[cmdParts.length - 1] = count.toString();             targetCmd = cmdParts.join(' ');         } else {             return;         }          const buttonText = `扫荡${count}次`;         let targetSpan = container.querySelector(`span[cmd="${targetCmd}"]`);         if (targetSpan) {             targetSpan.textContent = buttonText;         } else {             targetSpan = document.createElement('span');             targetSpan.setAttribute('cmd', targetCmd);             targetSpan.textContent = buttonText;             container.appendChild(targetSpan);         }     }      function handleContainer(container) {         const spans = container.querySelectorAll('span[cmd]');         let hasSweepOnce = false;         for (let i = 0; i < spans.length; i++) {             if (spans[i].textContent.trim() === '扫荡一次') {                 hasSweepOnce = true;                 break;             }         }         if (hasSweepOnce) {             updateContainer(container);         }     }      const initialContainers = document.querySelectorAll(containerSelector);     initialContainers.forEach(handleContainer);      const observer = new MutationObserver(mutations => {         mutations.forEach(mutation => {             mutation.addedNodes.forEach(node => {                 if (node.nodeType !== 1) return;                 if (node.matches(containerSelector)) {                     handleContainer(node);                 }                 const containers = node.querySelectorAll(containerSelector);                 containers.forEach(handleContainer);             });         });     });      observer.observe(document.body, {         childList: true,         subtree: true     });  } catch (error) {} </script>
